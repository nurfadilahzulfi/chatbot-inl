# ΓöÇΓöÇ Parse Indonesian number format ΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇ
def parse_indonesian_number(series: pd.Series) -> pd.Series:
    """Convert '1.135,75' ΓåÆ 1135.75 (Indonesian locale format)."""
    return (
        series.astype(str)
        .str.replace('.', '', regex=False)   # Remove thousands separator
        .str.replace(',', '.', regex=False)  # Replace decimal comma
        .str.replace('%', '', regex=False)   # Remove percent sign
        .pipe(pd.to_numeric, errors='coerce')
    )

# ΓöÇΓöÇ Load raw data ΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇ
raw = pd.read_csv(csv_path)

# Rename to standard English column names
raw.columns = ['Date', 'Close', 'Open', 'High', 'Low', 'Volume', 'Pct_Change']

# Parse date (Indonesian DD/MM/YYYY format)
raw['Date'] = pd.to_datetime(raw['Date'], format='%d/%m/%Y', dayfirst=True)

# Parse numeric columns
for col in ['Close', 'Open', 'High', 'Low', 'Pct_Change']:
    raw[col] = parse_indonesian_number(raw[col])

# Sort chronologically (oldest ΓåÆ newest)
raw = raw.sort_values('Date').reset_index(drop=True)
raw.drop(columns=['Volume'], inplace=True)  # All null ΓåÆ drop

print(f'≡ƒôè Dataset Shape   : {raw.shape}')
